START

    STORE question as
        [
        1. 5 +3
        2. 6+ 7
        3. 9 x 12
        4. 6 - 2
        5. 10 / 5
        ]

    STORE option as
        [
        1. A) 4 B) 6 C) 8 D) 4
        2. A) 25 B) 13 C) 8 D) 5
        3. A) 18 B) 6 C) 89 D) 2
        4. A) 4 B) 3 C) 1 D) 0
        5. A) 4 B) 8  C) 8 D) 2
        ]

     STORE correctanswer as
        [
        1) C
        2) B
        3) A
        4) A
        5) D
        ]

     SET score = 0
     SET currentquestion = 0
     SET totalquestion = 5

     OUTPUT "Hello,Lets start the quiz!"

     WHILE currentquestion < totalquestion
        OUTPUT question[currentquestion]
        OUTPUT option[currentquestion]

        INPUT guess

            IF guess = correctanswer[currentquestion]
                SET score = score + 1
                OUTPUT "That was correct.Lets move on to the next question."
            ELSE
            OUTPUT "That was incorrect.Lets move on to the next question."
            ENDIF

        SET currentquestion = currentquestion + 1
    ENDWHILE

    OUTPUT "Your final score is" + score + "out of 5"

    SET percentage = (score / totalquestion) * 100

    OUTPUT "Your percentage score is" + percentage + "%"

END
