Start
    INPUT score

    IF score >= 70
        OUTPUT "Distinction"

    ELSE IF score > 40 AND score < 69
        OUTPUT "Pass"

    ELSE IF score < 40
        OUTPUT "FAIL"

    ENDIF
End
